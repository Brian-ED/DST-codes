Move to the other shard quickly. Extra features: migrate(3, AllPlayers[3]) takes the second joined player to the other third connection. By default it takes MasterSim (world owner) to the other shard's portal (cause 0 is always an invalid id).
----
function migrate(id, p)
    for k,v in pairs(Ents) do
        if v.prefab == "cave_entrance" or v.prefab == "cave_exit" then
            local wm = v.components.worldmigrator
            if wm.id == id or 1 then
                TheWorld:PushEvent("ms_playerdespawnandmigrate", { player = p or ThePlayer, portalid = id or 0, worldid = wm.linkedWorld })
            end
        end
    end
end

AddClassPostConstruct("screens/playerhud", function(self)
    local redundantpause = false
    local redundantunpause = false
    local OnUpdate_new = self.OnUpdate
    self.OnUpdate = function(...)
        OnUpdate_new(...)
        if self.serverpaused then
            if not redundantpause then
                ThePlayer:PushEvent("multiplayerpaused")
                redundantpause = true
                redundantunpause = false
            end
        else
            if not redundantunpause then
                ThePlayer:PushEvent("multiplayerunpaused")
                redundantunpause = true
                redundantpause = false
            end
        end
        return
    end
end)
